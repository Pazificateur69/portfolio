import { useState, useEffect } from 'react';
import { ThreatAnalyzer } from '@/lib/security';

export const SecurityDashboard = () => {
  const [threats, setThreats] = useState<Threat[]>([]);
  const analyzer = new ThreatAnalyzer();

  useEffect(() => {
    const stream = analyzer.subscribe({
      onThreat: (t) => setThreats(prev => [...prev, t]),
      severity: 'critical',
    });
    return () => stream.unsubscribe();
  }, []);

  return <ThreatGrid data={threats} />;
};

import asyncio
from scanner import VulnEngine, Report

async def scan_target(host: str) -> Report:
    """Run full vulnerability assessment."""
    engine = VulnEngine(
        target=host,
        modules=['xss', 'sqli', 'rce'],
        threads=50,
    )
    results = await engine.execute()

    for vuln in results.critical():
        await vuln.notify(channel='#security')

    return results.to_report()
